<script setup>
import formatGenres from '~/utils/formatGenres'
 
const route = useRoute()
const movieId = route.params.id
 
const { data: movie, pending, error } = await useFetch(`/api/movie/${movieId}`)
 
console.log(movie.value.recommendations.results);
 
const getFullImageUrl = (path) => path ? `https://image.tmdb.org/t/p/w1280${path}` : 'https://via.placeholder.com/1280x720'
 
const formatActors = () => {
  return movie.value.credits.cast.map(actor => actor.name).slice(0, 12).join(', ')
}
</script>
 
<template>
  <div>
    <div v-if="pending">Načítám detail filmu...</div>
    <div v-else-if="error">Ups! Film se nepodařilo najít.</div>
 
    <div v-else-if="movie" class="relative">
      <div class="h-100 w-full relative">
        <img
            :src="getFullImageUrl(movie.backdrop_path)"
            :alt="movie.title"
            class="w-full h-full object-cover brightness-55 contrast-120"
        />
        <div class="absolute bottom-0 left-0 p-8 text-white">
          <h1 class="text-4xl font-bold">{{ movie.title }}</h1>
          <p class="italic text-lg opacity-80">{{ movie.tagline }}</p>
        </div>
      </div>
 
      <div class="max-w-4xl mx-auto p-8 grid md:grid-cols-3 gap-8">
        <img
            :src="`https://image.tmdb.org/t/p/w500${movie.poster_path}`"
            :alt="movie.title"
            class="rounded-xl shadow-2xl -mt-20 relative z-10"
        />
        <div class="md:col-span-2">
          <p class="text-sm text-lime-400 mb-4">{{ formatGenres(movie) }}</p>
          <p class="text-sm text-gray-300 mb-5">Hrají: {{ formatActors() }}</p>
          <p class="text-gray-200 text-lg font-light pb-6">{{ movie.overview }}</p>
 
          <div class="mt-6 flex gap-4">
            <div class="bg-gray-700 px-3 py-1 rounded font-bold">
              ★ {{ movie.vote_average.toFixed(1) }}
            </div>
            <div class="text-gray-500">
              {{ movie.release_date }} • {{ movie.runtime }} min
            </div>
          </div>
        </div>
      </div>
    </div>
  </div>
</template>
